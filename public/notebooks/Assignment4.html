def graph_identification():
    
    from statistics import mean, median
    
    degrees = []
    clusters = []
    
    for G in P1_Graphs:
        
        # set of degrees of the nodes in each graph
        print(G.degree())
        degrees_set = sorted(set(G.degree().values()))
        degrees.append(len(degrees_set))
        
        # clustering coefficient
        clustering = nx.average_clustering(G) 
        clusters.append(clustering)
                
        #print(len(degrees_set))
        #print(clustering)
        
    
    deg_mean = mean(degrees)
    clust_mean = mean(clusters)
    
    #print('\nmean number of degrees: ', deg_mean)
    #print('mean clustering coef.: ', clust_mean)
    
    # use mean values of #degrees and clustering coefficients to classify graphs
    algos = []
    for i in range(len(P1_Graphs)):
        
        # if there are nodes of G with high degree classify it as PA
        if degrees[i] > deg_mean:
            algos.append('PA')
            
        # otherwise classify as SW
        else:
            # use clustering coefficient to classify as low or high rewiring prob.
            if clusters[i] > clust_mean:
                algos.append('SW_L')
            else:
                algos.append('SW_H')

    return algos

def salary_predictions():
    
    nodes_df = pd.DataFrame(index=G.nodes())
    
    # define node features including the department number and centrality measures
    nodes_df['Department'] = pd.Series(nx.get_node_attributes(G,'Department'))
    nodes_df['Degree'] = pd.Series(nx.degree(G))
    nodes_df['ClusterCoef'] = pd.Series(nx.clustering(G))
    nodes_df['DegreeCent'] = pd.Series(nx.degree_centrality(G))
    nodes_df['ClosenessCent'] = pd.Series(nx.closeness_centrality(G))
    nodes_df['BtwnCent'] = pd.Series(nx.betweenness_centrality(G))
    
    features = nodes_df.columns.values.tolist()
    
    # add target variable as column then split data into train and test sets
    nodes_df['isMngmt'] = pd.Series(nx.get_node_attributes(G,'ManagementSalary'))
    
    train = nodes_df[~nodes_df['isMngmt'].isnull()].copy()
    test = nodes_df[nodes_df['isMngmt'].isnull()][features].copy()
    
    scaler = MinMaxScaler()
    
    X_train = scaler.fit_transform(train[features])
    X_test = scaler.transform(test)
    
    grid_params = {'C': [0.1, 1, 10, 100, 1000],
              'gamma': [1, 0.1, 0.01, 0.001, 0.0001],
              'kernel': ['rbf', 'linear']}
    
    # try SVC model
    svc_grid = GridSearchCV(SVC(probability=True), grid_params, scoring='roc_auc', refit = True)
    svm_clf = svc_grid.fit(X_train, train['isMngmt'].astype(int))
        
    prob = svm_clf.predict_proba(X_test)
    #print(prob)
    
    return pd.Series([x[1] for x in prob], index=test.index.values)
